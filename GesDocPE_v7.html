📖 Guía — Google Apps Script
Configura el Apps Script para conectar con tu Google Sheet y Drive
1
Abre tu Google Sheet y entra al editor de scripts
En el Sheet: Extensiones → Apps Script. Se abre el editor de código.
2
Pega este código completo
Borra todo el contenido del editor y reemplázalo con:
function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(d.sheet_id);
    var hoja = d.hoja || 'INF. EMIT.';
    var sheet = ss.getSheetByName(hoja);
    
    // Crear hoja si no existe
    if (!sheet) {
      sheet = ss.insertSheet(hoja);
      sheet.appendRow(['NOMBRE REMITENTE','N° INF. REMITENTE',
        'N° INF.','ASUNTO','N° FOLIOS','FECHA',
        'OBRA','REMITIDO A','OBSERVACIONES']);
    }
    
    var lastRow = sheet.getLastRow();
    var ultimoNum = 0;
    
    // Buscar el último número correlativo
    if (lastRow > 1) {
      var vals = sheet.getRange(2, 3, lastRow-1, 1).getValues();
      vals.forEach(function(r) {
        var n = parseInt(r[0]);
        if (!isNaN(n) && n > ultimoNum) ultimoNum = n;
      });
    }
    
    // Verificar si el número ya existe
    var numInf = parseInt(d.num_informe) || (ultimoNum + 1);
    if (d.num_informe && d.num_informe != '' ) {
      var yaExiste = false;
      if (lastRow > 1) {
        var existentes = sheet.getRange(2,3,lastRow-1,1).getValues();
        existentes.forEach(function(r){ if(parseInt(r[0])==numInf) yaExiste=true; });
      }
      if (yaExiste) numInf = ultimoNum + 1;
    }
    
    var fecha = Utilities.formatDate(new Date(),'America/Lima','dd/MM/yyyy');
    
    // Registrar — SIN observaciones (columna I vacía)
    sheet.appendRow([
      d.nombre_remitente || '',
      d.num_inf_remitente || '',
      numInf,
      d.asunto || '',
      d.num_folios || '2',
      fecha,
      d.nombre_obra || d.clave || '',
      d.destinatario || 'GM',
      '' // Observaciones vacías — el usuario las llena
    ]);
    
    return ContentService
      .createTextOutput(JSON.stringify({ok:true, numInf:numInf, siguiente: numInf+1}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false, error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  // Para consultar el último número
  try {
    var ss = SpreadsheetApp.openById(e.parameter.sheet_id);
    var sheet = ss.getSheetByName(e.parameter.hoja || 'INF. EMIT.');
    if (!sheet) return ContentService.createTextOutput(JSON.stringify({ultimoNum:0})).setMimeType(ContentService.MimeType.JSON);
    var lastRow = sheet.getLastRow();
    var ultimoNum = 0;
    if (lastRow > 1) {
      var vals = sheet.getRange(2,3,lastRow-1,1).getValues();
      vals.forEach(function(r){ var n=parseInt(r[0]); if(!isNaN(n)&&n>ultimoNum) ultimoNum=n; });
    }
    return ContentService.createTextOutput(JSON.stringify({ultimoNum:ultimoNum, siguiente:ultimoNum+1})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ultimoNum:0,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
